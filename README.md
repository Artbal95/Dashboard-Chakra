## Install

```bash
npm install --save dashboard-chakra
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'dashboard-chakra'
import 'dashboard-chakra/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
